void JobSystem::WorkerThreadLoop()
{
    while(true) {
        // 使用条件变量等待新任务或关闭信号
        wakeWorker.wait(lock, [&]{ return !isRunning || !jobQueue.empty();});
        if(!isRunning && jobQueue.empty()) return;
        if(jobQueue.empty()) continue;
        job = jobQueue.front();
        jobQueue.pop_front();
        job.function(job.JobData, job.rawCounter);
    }
}
